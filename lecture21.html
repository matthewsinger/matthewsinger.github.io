; DATA DEFINITIONS ;;;;;;;;;;
 
(define-struct snake [dir segs])
(define-struct world [snake food])
; A World is (make-world Snake Food)
; Interpretation: The world state of our game
; - snake is the current snake
; - food is the location of the current food
; Origin is at top left
 
; A Food is a (make-posn Number Number)
; Interpretation: The location of the food
; - x is the x-coordinate IN THE GRID of the food
; - y is the y-coordinate IN THE GRID of the food
 
; A Snake is a (make-snake Direction Segs)
; Interpretation: The current state of the snake
; - dir is the direction the snake is moving
; - segs is the location of the body segments of the snake
; Snake must contain at least one segment
; The head is the first element in segs
 
; A Segs is one of:
; - '()
; - (cons Posn Segs)
; Interpretation: A list of body segments
; - x is the x-coordinate IN THE GRID of the segment
; - y is the y-coordinate IN THE GRID of the segment
 
; A Direction is one of:
; - "up"
; - "down"
; - "left"
; - "right"
; Interpretation: The direction of movement of the snake
 
; CONSTANTS ;;;;;;;
 
(define GRID-SQSIZE 10) ; the width and height of the grid squares
(define BOARD-HEIGHT 20) ; the hieght of the game board in grid squares
(define BOARD-WIDTH 30) ; the width of the gram eboard in grid squares
(define BOARD-HEIGHT/PIX (* BOARD-HEIGHT GRID-SQSIZE)) ; board height in pixels
(define BOARD-WIDTH/PIX (* BOARD-WIDTH GRID-SQSIZE)) ; board width in pixels
 
(define BACKGROUND (empty-scene BOARD-WIDTH/PIX BOARD-HEIGHT/PIX))
 
(define SEGMENT-RADIUS (/ GRID-SQSIZE 2))
(define SEGMENT-IMAGE (circle SEGMENT-RADIUS "solid" "red"))
 
(define FOOD-RADIUS (floor (* 0.9 SEGMENT-RADIUS)))
(define FOOD-IMAGE (circle FOOD-RADIUS "solid" "green"))
 
(define TICK-RATE 0.3)

(define segs1 '())

; world->scene : World -> Image
; Render the game state as an image
 
; snake+scene : Snake Image -> Image    ;; Done!
 
; food+scene : Food Image -> Image
 
; key-handler : World KeyEvent -> World
; Changes the direction of the snake; leaves everytging else alone
 
; world-done? : World -> Boolean
; Checks whether we hit the wall or we hit the snake
 
; world->world : World -> World
; Checks if snake is about to eat the food
;    if so, make a new food AND add a new head for the snake
;    if not, make a new head for the snake AND drop the final segment
 
; eating? :
; snake-eat
; snake-grow
; snake-move
 
; 
; Image/scene rendering functions
; 
 
; world->scene : World -> Image
; Render the game state as an image
(define (world->scene w)
  (snake+scene (world-snake w)
               (food+scene (world-food w) BACKGROUND)))
 
; food+scene : Food Image -> Image
; Add image of food to the given scene
(define (food+scene f img)
  (place-image-on-grid FOOD-IMAGE (posn-x f) (posn-y f) img))
 
; snake+scene : Snake Image -> Image
; Draws a snake onto the provided image
(define (snake+scene s image)
  (segs+scene (snake-segs s) image))
 
(check-expect (snake+scene snake0 BACKGROUND)
              (segs+scene segs3 BACKGROUND))
 
; segs+scene : Segs Image -> Image
; Draws an entire snake onto the provided image
(define (segs+scene s img)
  (cond
    [(empty? s) img]
    [(cons? s) (seg+scene (first s) (segs+scene (rest s) img))]))
 
(check-expect (segs+scene segs3 BACKGROUND)
              (seg+scene (make-posn 3 5) (seg+scene (make-posn 2 5) BACKGROUND)))
 
; seg+scene : Posn Image -> Image
; Draws a single snake segment onto the provided image
(define (seg+scene p img)
  (place-image-on-grid SEGMENT-IMAGE (posn-x p) (posn-y p) img))
 
(check-expect (seg+scene (make-posn 3 4) BACKGROUND)
              (place-image-on-grid SEGMENT-IMAGE 3 4 BACKGROUND))
 
; place-image-on-grid : Image Number Number Image -> Image
; Places an image at the grid locations x and y on top of the second image
(define (place-image-on-grid img1 x y img2)
  (place-image img1
               (+ (* x GRID-SQSIZE) (/ GRID-SQSIZE 2))
               (+ (* y GRID-SQSIZE) (/ GRID-SQSIZE 2))
               img2))
 
(check-expect (place-image-on-grid (circle 20 "solid" "yellow") 3 4 BACKGROUND)
              (place-image (circle 20 "solid" "yellow") 35 45 BACKGROUND))

(check-expect (segs+scene (cons (make-posn 6 10) empty) BACKGROUND)
              (place-image SEGMENT-IMAGE 65 105 BACKGROUND))
 
(check-expect (snake+scene snake1 BACKGROUND)
              (place-image SEGMENT-IMAGE 65 105 BACKGROUND))
 
(check-expect (food+scene food1 BACKGROUND)

(check-expect (world->scene world1)

; 
; World to next World on tick
; 
; world->world : World -> World
; change the world one step
; (Used version 1 for early testing)
(define (world->world.v1 w)
  (make-world (snake-slither (world-snake w))
              (world-food w)))
 
(define (world->world w)
  (cond [(eating? w) (eat&grow w)]
        [else (make-world (snake-slither (world-snake w))
                          (world-food w))]))
 
; 
; Snake motion and growth
; 
 
; snake-slither : Snake -> Snake
; Move the snake by one step in approp direction
;  How : new head is old head moved over by one step in approp direc
;        new tail is old segs minus last
(define (snake-slither snk)
  (make-snake (snake-dir snk)
              (move-segs (snake-segs snk) (snake-dir snk))))
 
; A NESegs is one of:
; - (cons Posn empty)
; - (cons Posn NESegs)
 
 
 
 
 
; move-segs : NESegs Direction -> NESegs
; Move the snake's segments
;  How : new head is old head moved over by one step in approp direc
;        new tail is old segs minus last
(define (move-segs nesegs dir)
  (cons (move-seg (first nesegs) dir)
        (segments-all-but-last nesegs)))
 
; segments-all-but-last : NESegs -> Segs
; remove the last segment
(define (segments-all-but-last nesegs)
  (cond [(empty? (rest nesegs)) empty]
        [else  (cons (first nesegs) (segments-all-but-last (rest nesegs)))]))
 
 
; move-seg : Posn Direction -> Posn
; move the segment in the given direction
(define (move-seg p d)
  (cond
    [(string=? d "up") (make-posn (posn-x p) (sub1 (posn-y p)))]
    [(string=? d "down") (make-posn (posn-x p) (add1 (posn-y p)))]
    [(string=? d "left") (make-posn (sub1 (posn-x p)) (posn-y p))]
    [(string=? d "right") (make-posn (add1 (posn-x p)) (posn-y p))]))
 
; ——————————
; eat&grow : World -> World
; Eat the current food, grow the snake and produce new food
(define (eat&grow w)
  (make-world (snake-grow (world-snake w))
              (make-posn (add1 (random BOARD-WIDTH))
                         (add1 (random BOARD-HEIGHT)))))
 
; snake-grow : Snake -> Snake
; Grow snake one step
; This is just like snake-slither except we don't drop last seg
(define (snake-grow snk)
  (make-snake (snake-dir snk)
              (cons (move-seg (first (snake-segs snk)) (snake-dir snk))
                    (snake-segs snk))))
 
 
; ———————————————
; Examples/Tests for snake movement and growth
 
(check-expect (segments-all-but-last
               (cons (make-posn 6 10) empty))
              empty)
 
(check-expect (segments-all-but-last
               (cons (make-posn 5 10)
                     (cons (make-posn 6 10) empty)))
              (cons (make-posn 5 10) empty))
 
(check-expect (move-seg (make-posn 6 10) "up")
              (make-posn 6 9))
(check-expect (move-seg (make-posn 6 10) "down")
              (make-posn 6 11))
(check-expect (move-seg (make-posn 6 10) "left")
              (make-posn 5 10))
(check-expect (move-seg (make-posn 6 10) "right")
              (make-posn 7 10))
 
(check-expect (snake-slither snake1)
              (make-snake "left"
                          (cons (make-posn 5 10) empty)))
 
(check-expect (snake-grow snake1)
              (make-snake "left"
                          (cons (make-posn 5 10)
                                (cons (make-posn 6 10) empty))))
 
; 
; Key handler
; 
; key-handler : World Key-Event -> World
; handle things when the user hits a key on the keyboard.
(define (key-handler w ke)
  (cond [(key=? ke "n") world1] ; New game
        [(or (key=? ke "up")
             (key=? ke "down")
             (key=? ke "left")
             (key=? ke "right"))
         (make-world (make-snake ke (snake-segs (world-snake w)))
                     (world-food w))]
        [else w]))
 
 
; 
; Collision detection
; 
 
; posn=? : Posn Posn -> Boolean
; Are the two posns equal?
(define (posn=? p1 p2)
  (and (= (posn-x p1) (posn-x p2))
       (= (posn-y p1) (posn-y p2))))
 
; eating? : World -> Boolean
; check if snake head overlaps food
(define (eating? w)
  (posn=? (first (snake-segs (world-snake w)))
          (world-food w)))
 
; 
; world-done? : World -> Boolean
; has the game ended?
(define (world-done? w)
  (snake-death? (world-snake w)))
 
; snake-death? : Snake -> Boolean
; Has the snake self-collided or wall-collided?
(define (snake-death? snk)
  (or (self-collide? snk)
      (wall-collide? (first (snake-segs snk)))))
 
; self-collide? : Snake -> Boolean
; Does the snake collide with itself?
(define (self-collide? snk)
  (segs-collide? (first (snake-segs snk))
                 (rest  (snake-segs snk))))
 
; segs-collide? : Posn Segs -> Boolean
; Does head posn equal any of the posns in the list Segs?
(define (segs-collide? head segs)
  (and (not (empty? segs))
       (or (posn=? (first segs) head)
           (segs-collide? head (rest segs)))))
 
; wall-collide? : Posn -> Boolean
; Does the head of the snake collide with a wall?
(define (wall-collide? head)
  (or (<= (posn-x head) 0)
      (> (posn-x head) BOARD-WIDTH)
      (<= (posn-y head) 0)
      (> (posn-y head) BOARD-HEIGHT)))
 
 
; Examples/Tests for collision functions
(check-expect (eating? world2) #true)
(check-expect (eating? world3) #false)
 
 
 
(big-bang world1
  (to-draw world->scene)
  (on-tick world->world TICK-RATE)
  (on-key key-handler)
  (stop-when world-done?))
